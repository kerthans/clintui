## Next Themes
Because you are using Next.js, you can use `next-themes` to implement dark mode. And since you're using `@intentui/icons`, you can use the `IconSun` and `IconMoon` icons.

## Usage
After that, you can use it in your root layout file like so:

```tsx {1} {7-9}
import { Providers } from '@/components/providers'

export default function RootLayout({ children }: Readonly<{ children: React.ReactNode }>) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <Providers>
          {children}
        </Providers>
      </body>
    </html>
  )
}
```
